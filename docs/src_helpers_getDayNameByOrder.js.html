/**
 * Converts a day number to a 'd' string.
 *@function
 * @param {Number} dayIndex
 * @return {String} Returns day as string
 */

export const GetDayNameByOrder = (tickItem) => {
  if (tickItem === 1) {
    return "L";
  }
  if (tickItem === 2) {
    return "M";
  }
  if (tickItem === 3) {
    return "M";
  }
  if (tickItem === 4) {
    return "J";
  }
  if (tickItem === 5) {
    return "V";
  }
  if (tickItem === 6) {
    return "S";
  }
  if (tickItem === 7) {
    return "D";
  }
};
